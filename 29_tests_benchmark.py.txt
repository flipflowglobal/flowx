def bench(name, fn, n):
    # Warmup
    fn()
    t0 = time.perf_counter()
    for _ in range(n):
        fn()
    elapsed = time.perf_counter() - t0
    per_call = elapsed / n * 1000
    print(f"  {name:30s}  {per_call:8.3f} ms/call  ({n} iters, {elapsed:.2f}s total)")
    return per_call

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--n", type=int, default=50, help="Iterations per backend")
    args = parser.parse_args()
    N = args.n

    print()
    print("╔══════════════════════════════════════════════════════════╗")
    print("║         HINSDALE BENCHMARK — CYTHON vs SUBPROCESS        ║")
    print("╚══════════════════════════════════════════════════════════╝")
    print(f"  Bytecode: {len(AUREON_HEX)//2} bytes  |  Iterations: {N}")
    print()

    results = {}

    # ── 1. Cython (direct FFI) ────────────────────────────────────────────
    try:
        import _hinsdale as cy
        print(f"  Cython backend: {cy.version()}")

        def _cy_full():
            ctx = cy.analyze(AUREON_HEX)
            _ = ctx.pseudo_source()
            _ = ctx.functions()
            _ = ctx.findings()

        def _cy_disasm():
            ctx = cy.analyze(AUREON_HEX)
            _ = ctx.instructions_numpy()

        def _cy_risk():
            ctx = cy.analyze(AUREON_HEX)
            return ctx.risk_score()

        print()
        print("  [CYTHON — direct C FFI, GIL released]")
        results["cy_full"]   = bench("full analysis (Cython)",    _cy_full,   N)
        results["cy_disasm"] = bench("disasm+numpy (Cython)",      _cy_disasm, N)
        results["cy_risk"]   = bench("risk_score only (Cython)",   _cy_risk,   N)

    except ImportError:
        print("  [CYTHON] Not compiled. Run: python setup_cython.py build_ext --inplace")

    # ── 2. Subprocess JSON ────────────────────────────────────────────────
    try:
        from hinsdale import Hinsdale
        h = Hinsdale()
        if h._using_rust:
            def _sub_full():
                r = h.analyze(AUREON_HEX)
                _ = r.pseudo_source
                _ = r.functions
                _ = r.findings
            print()
            print("  [SUBPROCESS — JSON pipe to hinsdale-cli]")
            results["sub_full"] = bench("full analysis (subprocess)", _sub_full, min(N, 20))
        else:
            print("  [SUBPROCESS] hinsdale-cli binary not found")
    except Exception as e:
        print(f"  [SUBPROCESS] Error: {e}")

    # ── 3. Pure Python disasm ─────────────────────────────────────────────
    try:
        from hinsdale import _py_disassemble, _py_fallback
        raw = bytes.fromhex(AUREON_HEX)

        def _py_disasm():
            return _py_disassemble(raw)

        def _py_full():
            return _py_fallback(raw)

        print()
        print("  [PURE PYTHON — reference baseline]")
        results["py_disasm"] = bench("disasm (Python)",      _py_disasm, N)
        results["py_full"]   = bench("full fallback (Python)", _py_full, N)
    except Exception as e:
        print(f"  [PYTHON] Error: {e}")

    # ── Speedup summary ───────────────────────────────────────────────────
    print()
    print("  ── Speedup ratios ───────────────────────────────────────")
    if "cy_full" in results and "sub_full" in results:
        ratio = results["sub_full"] / results["cy_full"]
        print(f"  Cython vs Subprocess:  {ratio:.1f}x faster")
    if "cy_disasm" in results and "py_disasm" in results:
        ratio = results["py_disasm"] / results["cy_disasm"]
        print(f"  Cython vs Python disasm: {ratio:.1f}x faster")
    if "cy_full" in results and "py_full" in results:
        ratio = results["py_full"] / results["cy_full"]
        print(f"  Cython vs Python full:  {ratio:.1f}x faster")
    print()

if __name__ == "__main__":
    main()
